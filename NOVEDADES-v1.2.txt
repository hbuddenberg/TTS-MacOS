┌──────────────────────────────────────────────────────────────────────┐
│  🚀 INSTALACIÓN Y USO DE UVX                                         │
└──────────────────────────────────────────────────────────────────────┘

    Paso 1: Instalar uv (una sola vez)
    ──────────────────────────────────────────────────────────────
    brew install uv
    
    # O con script oficial:
    curl -LsSf https://astral.sh/uv/install.sh | sh
    

    Paso 2: Usar directamente (sin instalar el proyecto)
    ──────────────────────────────────────────────────────────────
    cd mcp-tts-macos
    
    # Básico
    uvx --from . tts-macos "Hola mundo"
    
    # Con opciones
    uvx --from . tts-macos "Buenos días" --voice jorge --rate 200
    
    # Guardar audio
    uvx --from . tts-macos "Mi mensaje" --save audio.aiff
    

    Paso 3: Crear alias para uso frecuente (opcional)
    ──────────────────────────────────────────────────────────────
    # Agregar a ~/.zshrc
    alias tts='uvx --from ~/mcp-tts-macos tts-macos'
    
    # Ahora es más corto
    tts "Hola mundo"
    tts "Buenos días" --voice jorge

┌──────────────────────────────────────────────────────────────────────┐
│  🎨 CASOS DE USO CON UVX                                             │
└──────────────────────────────────────────────────────────────────────┘

    1. Notificación de scripts:
    ──────────────────────────────────────────────────────────────
    ./backup.sh && uvx --from . tts-macos "Backup completado"
    

    2. Timer/Recordatorio:
    ──────────────────────────────────────────────────────────────
    sleep 1800 && uvx --from . tts-macos "Han pasado 30 minutos"
    

    3. Leer archivos:
    ──────────────────────────────────────────────────────────────
    uvx --from . tts-macos "$(cat documento.txt)" --voice jorge
    

    4. Pipeline con otros comandos:
    ──────────────────────────────────────────────────────────────
    cat noticias.txt | head -3 | xargs uvx --from . tts-macos
    

    5. Monitor de sistema:
    ──────────────────────────────────────────────────────────────
    while true; do
        cpu=$(top -l 1 | grep "CPU" | awk '{print $3}')
        if [ $cpu -gt 80 ]; then
            uvx --from . tts-macos "CPU al $cpu%"
        fi
        sleep 300
    done
    

    6. Cron (recordatorios):
    ──────────────────────────────────────────────────────────────
    # En crontab:
    0 9 * * * uvx --from ~/mcp-tts-macos tts-macos "Buenos días"

    Alias útiles para ~/.zshrc:
    ──────────────────────────────────────────────────────────────
    # Simple
    alias tts='uvx --from ~/mcp-tts-macos tts-macos'
    
    # Con voces
    alias monica='tts --voice monica'
    alias jorge='tts --voice jorge'
    
    # Función completa
    function hablar() {
        uvx --from ~/mcp-tts-macos tts-macos "$@"
    }
    

    Script de ejemplo:
    ──────────────────────────────────────────────────────────────
    #!/bin/bash
    # notificar.sh
    
    PROJECT_DIR=~/mcp-tts-macos
    
    function notificar() {
        uvx --from "$PROJECT_DIR" tts-macos "$1" --voice monica
    }
    
    # Uso
    ./mi_script.sh
    if [ $? -eq 0 ]; then
        notificar "Script completado exitosamente"
    else
        notificar "Error en el script"
    fi
